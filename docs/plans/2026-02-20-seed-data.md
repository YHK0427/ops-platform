```python
"""
DB 초기화 + 테스트용 시드 데이터 삽입
설계 문서: docs/plans/2026-02-20-seed-data-design.md

import asyncio
import os
import sys
from datetime import date, datetime, timezone

sys.path.append(os.path.join(os.path.dirname(__file__), ".."))

from sqlalchemy import text
from app.database import AsyncSessionLocal
from app.models import (
    Assignment, Attendance, Ledger, Member,
    Session as SessionModel, Team, TeamHistory, TeamMember,
)


# ────────────────────────────────────────────────
# 헬퍼
# ────────────────────────────────────────────────

def now_utc() -> datetime:
    return datetime.now(timezone.utc)


async def truncate_all(db) -> None:
    print("  [1/6] 테이블 초기화 중...")
    tables = [
        "ledger", "team_history", "assignments", "attendance",
        "team_members", "teams", "sessions", "members", "naver_sessions",
    ]
    for t in tables:
        await db.execute(text(f"TRUNCATE TABLE {t} RESTART IDENTITY CASCADE"))
    await db.commit()
    print("       완료")


# ────────────────────────────────────────────────
# 멤버 생성
# ────────────────────────────────────────────────

async def create_members(db) -> list[Member]:
    print("  [2/6] 멤버 생성 중...")

    # (name, email, deposit, minus_score)
    # minus_score는 week1 FINALIZE 결과가 이미 반영된 값
    # 오도현(m7)은 week1 이전 사전 점수 -9 → week1 결과 -5 추가 → 합계 -14
    member_specs = [
        ("김민준", "minjun@example.com",  20_000,  0),
        ("이서연", "seoyeon@example.com", 17_000, -1),  # -3000 차감
        ("박지훈", "jihoon@example.com",  12_000, -4),  # -8000 차감
        ("최수아", "sooah@example.com",   18_000, -1),  # -2000 차감
        ("정우진", "woojin@example.com",  20_000,  0),
        ("강예린", "yerin@example.com",   19_000, -1),  # -1000 차감
        ("오도현", "dohyun@example.com",   4_000, -14), # -9(prior) -4-1 = -14, -8000-1000-5000(milestone)
        ("윤지아", "jia@example.com",     14_000, -4),  # -6000 차감
        ("임준서", "junseo@example.com",  20_000,  0),
        ("한소희", "sohee@example.com",   16_000, -1),  # -4000 차감
    ]

    members = []
    for name, email, deposit, minus in member_specs:
        m = Member(
            name=name,
            email=email,
            current_deposit=deposit,
            total_plus_score=0,
            total_minus_score=minus,
            net_score=minus,          # net = plus(0) + minus
            is_active=True,
        )
        db.add(m)
        members.append(m)

    await db.commit()
    # ID 갱신을 위해 refresh
    for m in members:
        await db.refresh(m)

    print(f"       {len(members)}명 생성 완료")
    return members


# ────────────────────────────────────────────────
# Week 1 — FINALIZED (TEAM)
# ────────────────────────────────────────────────

async def create_week1(db, members: list[Member]) -> SessionModel:
    print("  [3/6] Week 1 (FINALIZED TEAM) 생성 중...")

    m1, m2, m3, m4, m5, m6, m7, m8, m9, m10 = members

    s = SessionModel(
        week_num=1,
        title="OT & 팀 킥오프",
        date=date(2026, 2, 6),
        type="TEAM",
        status="FINALIZED",
        finalized_at=datetime(2026, 2, 6, 14, 0, 0, tzinfo=timezone.utc),
        config={"has_ppt": True, "has_review": True, "has_feedback": True, "is_holiday": False},
    )
    db.add(s)
    await db.commit()
    await db.refresh(s)

    # ── 팀 구성 ──────────────────────────────────
    team_a = Team(session_id=s.id, name="Team A")
    team_b = Team(session_id=s.id, name="Team B")
    team_c = Team(session_id=s.id, name="Team C")
    db.add_all([team_a, team_b, team_c])
    await db.commit()
    for t in [team_a, team_b, team_c]:
        await db.refresh(t)

    # Team A: m1(leader), m2, m3, m4
    for m, leader in [(m1, True), (m2, False), (m3, False), (m4, False)]:
        db.add(TeamMember(team_id=team_a.id, member_id=m.id, is_leader=leader))
    # Team B: m5(leader), m6, m7, m8
    for m, leader in [(m5, True), (m6, False), (m7, False), (m8, False)]:
        db.add(TeamMember(team_id=team_b.id, member_id=m.id, is_leader=leader))
    # Team C: m9(leader), m10
    for m, leader in [(m9, True), (m10, False)]:
        db.add(TeamMember(team_id=team_c.id, member_id=m.id, is_leader=leader))

    await db.commit()

    # ── 출결 ─────────────────────────────────────
    # (member, status, excuse_type)
    att_specs = [
        (m1, "PRESENT",      None),
        (m2, "LATE_UNDER10", "POST"),
        (m3, "ABSENT",       None),
        (m4, "LATE_UNDER10", "PRE"),
        (m5, "EXCUSED",      None),
        (m6, "PRESENT",      None),
        (m7, "ABSENT",       None),
        (m8, "ABSENT",       "POST"),
        (m9, "PRESENT",      None),
        (m10,"LATE_UNDER10", None),
    ]
    for m, status, excuse in att_specs:
        db.add(Attendance(session_id=s.id, member_id=m.id, status=status, excuse_type=excuse))

    # ── 과제 ─────────────────────────────────────
    # PPT: team_id 기반, member_id=None (BUG-04 현실 반영)
    db.add(Assignment(session_id=s.id, team_id=team_a.id, member_id=None, type="PPT", status="PASS"))
    db.add(Assignment(session_id=s.id, team_id=team_b.id, member_id=None, type="PPT", status="MISSING"))
    db.add(Assignment(session_id=s.id, team_id=team_c.id, member_id=None, type="PPT", status="PASS"))

    # REVIEW / FEEDBACK: member별
    # 강예린(m6): REVIEW MISSING
    # 오도현(m7): REVIEW MISSING, FEEDBACK MISSING
    # 나머지: PASS
    for m in members:
        review_status = "MISSING" if m in (m6, m7) else "PASS"
        feedback_status = "MISSING" if m == m7 else "PASS"
        db.add(Assignment(session_id=s.id, member_id=m.id, type="REVIEW",   status=review_status))
        db.add(Assignment(session_id=s.id, member_id=m.id, type="FEEDBACK", status=feedback_status))

    await db.commit()

    # ── 원장 (FINALIZE 결과) ─────────────────────
    finalized_at = s.finalized_at

    def add_ledger(member, ltype, amount, score, desc, deposit_after):
        db.add(Ledger(
            session_id=s.id,
            member_id=member.id,
            type=ltype,
            amount_krw=amount,
            score_delta=score,
            description=desc,
            deposit_after=deposit_after,
            created_at=finalized_at,
        ))

    # 이서연: LATE_UNDER10/POST → -1점, -3,000원
    add_ledger(m2, "FINE", -3_000, -1, "LATE_UNDER10 (POST)", 17_000)
    # 박지훈: ABSENT → -4점, -8,000원
    add_ledger(m3, "FINE", -8_000, -4, "ABSENT (사유서없음)", 12_000)
    # 최수아: LATE_UNDER10/PRE → -1점, -2,000원
    add_ledger(m4, "FINE", -2_000, -1, "LATE_UNDER10 (PRE)", 18_000)
    # 강예린: REVIEW MISSING → -1점, -1,000원
    add_ledger(m6, "FINE", -1_000, -1, "미제출: REVIEW", 19_000)
    # 오도현: ABSENT → -4점, -8,000원
    add_ledger(m7, "FINE", -8_000, -4, "ABSENT (사유서없음)", 12_000)
    # 오도현: REVIEW+FEEDBACK MISSING → -1점, -1,000원
    add_ledger(m7, "FINE", -1_000, -1, "미제출: REVIEW, FEEDBACK", 11_000)
    # 오도현: MILESTONE_FINE -10점 도달 → -5,000원
    add_ledger(m7, "MILESTONE_FINE", -5_000, 0, "누적벌점 -10점 도달 추가 벌금", 4_000)  # 6000 intermediate, but milestone reduces to 4000 (net after both deductions + milestone)
    # 주의: 위 deposit_after는 최종 상태(4,000)로 맞춘다.
    # 윤지아: ABSENT/POST → -4점, -6,000원
    add_ledger(m8, "FINE", -6_000, -4, "ABSENT (POST)", 14_000)
    # 한소희: LATE_UNDER10 → -1점, -4,000원
    add_ledger(m10, "FINE", -4_000, -1, "LATE_UNDER10 (사유서없음)", 16_000)

    await db.commit()

    # ── Team History ──────────────────────────────
    from itertools import combinations
    all_teams = [
        [m.id for m in [m1, m2, m3, m4]],  # Team A
        [m.id for m in [m5, m6, m7, m8]],  # Team B
        [m.id for m in [m9, m10]],          # Team C
    ]
    for team_member_ids in all_teams:
        for a, b in combinations(sorted(team_member_ids), 2):
            db.add(TeamHistory(session_id=s.id, member_a_id=a, member_b_id=b))
    await db.commit()

    print("       Week 1 완료")
    return s

        review_status  = "MISSING" if m in (m5, m9) else "PASS"

        db.add(Assignment(session_id=s.id, member_id=m.id, type="REVIEW",   status=review_status))
        db.add(Assignment(session_id=s.id, member_id=m.id, type="FEEDBACK", status=feedback_status))

async def seed():
    print("\n🌱 Seeding started...\n")
    async with AsyncSessionLocal() as db:
        await truncate_all(db)
        members = await create_members(db)
        await create_week1(db, members)
        await create_week2(db, members)
        await create_week3(db, members)
        await create_week4(db)

    print("\n✅ Seeding complete!\n")
    print("요약:")
    print("  멤버 10명 (김민준·이서연·박지훈·최수아·정우진·강예린·오도현·윤지아·임준서·한소희)")
    print("  Week 1 — FINALIZED TEAM  (원장 9건, TeamHistory 8건)")
    print("  Week 2 — SETTLEMENT TEAM (정산 프리뷰 테스트용)")
    print("  Week 3 — OPS INDIVIDUAL  (출결 입력 테스트용)")
    print("  Week 4 — SETUP INDIVIDUAL (팀편성 테스트용)")


if __name__ == "__main__":
    asyncio.run(seed())
```
